32// Arduino Wire library is required if I2Cdev I2CDEV_ARDUINO_WIRE implementation

36// I2Cdev and MPU6050 must be installed as libraries, or else the .cpp/.h files

41// class default I2C address is 0x68
42// specific I2C addresses may be passed as a parameter here
43// AD0 low = 0x68 (default for InvenSense evaluation board)
44// AD0 high = 0x69
45MPU6050 accelgyro;

48int16_t gx, gy, gz;

50#define LED_PIN 13
51bool blinkState = false;

53void setup() {
54 // join I2C bus (I2Cdev library doesn't do this automatically)
55 Wire.begin();
56
57 // initialize serial communication
58 // (38400 chosen because it works as well at 8MHz as it does at 16MHz, but
59 // it's really up to you depending on your project)
60 Serial.begin(38400);
61
62 // initialize device
63 while(!Serial);
64 Serial.println("Initializing I2C devices...");
65 accelgyro.initialize();
66
67 // verify connection
68 Serial.println("Testing device connections...");
69 Serial.println(accelgyro.testConnection() ? "MPU6050 connection successful" : "MPU6050 connection failed");
70
71 // configure Arduino LED for
72 pinMode(LED_PIN, OUTPUT);
73}
74
75void loop() {
76 // read raw accel/gyro measurements from device
77 accelgyro.getMotion6(&ax, &ay, &az, &gx, &gy, &gz);
78
79 // these methods (and a few others) are also available
80 //accelgyro.getAcceleration(&ax, &ay, &az);
81 //accelgyro.getRotation(&gx, &gy, &gz);
82
83 // display tab-separated accel/gyro x/y/z values
84 Serial.print("a/g:\t");
85 Serial.print(ax); Serial.print("\t");
86 Serial.print(ay); Serial.print("\t");
87 Serial.print(az); Serial.print("\t");
88 Serial.print(gx); Serial.print("\t");
89 Serial.print(gy); Serial.print("\t");
90 Serial.println(gz);
91
92 // blink LED to indicate activity
93 blinkState = !blinkState;
94 digitalWrite(LED_PIN, blinkState);
95}
