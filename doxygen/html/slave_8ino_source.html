75 static char message[MAX_MESSAGE_LENGTH];

77 static unsigned int message_pos = 0;

79 //Read the next available byte in the serial receive buffer
80 char inByte = blueToothSerial.read();
82 //Message coming in (check not terminating character)
83 if (inByte != '\n' && (message_pos < MAX_MESSAGE_LENGTH - 1)) {
84 //Add the incoming byte to our message
85 message[message_pos] = inByte;
86 message_pos++;
87 }
89 else {
90 //print the message
91 Serial.print("The message:");
92 //Print the message (or do other things)
93 Serial.println(message);

95 char dl1 = 0x11;
96 char dl2 = 0x12;
97 char indexC = 0x13;
98 //The roll value and picht value as strings
99 String r_v = "", p_v = "";
101 for (int i = 0; i < MAX_MESSAGE_LENGTH - 1; i++) {
102 char current = message[i];
103 if (current == 00) {
104 } else if (current == dl2 || current == dl1) {
105 indexC = current;
106 } else if (indexC == dl2) {
107 p_v += current;
108 } else if (indexC == dl1) {
109 r_v += current;
110 }
111 }
113 double roll = atof(r_v.c_str());
114 double picht = atof(p_v.c_str());
115 forward = 25.5 * picht;
116 turn = 180-(roll+10)*9;
117
118 //print the foward and turn values and the roll and picht values
119 Serial.println("turn: " + String(roll) +" , " + String(forward)+ " forward: " + String(picht)+" , " + String(turn));
121 message[message_pos] = '\0';
