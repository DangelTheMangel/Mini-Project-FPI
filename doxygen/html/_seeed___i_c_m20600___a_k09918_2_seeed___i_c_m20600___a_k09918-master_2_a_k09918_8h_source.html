67// #define AK09918_MEASURE_PERIOD 9 // Must not be changed
68// AK09918 has following seven operation modes:
69// (1) Power-down mode: AK09918 doesn't measure
70// (2) Single measurement mode: measure when you call any getData() function
71// (3) Continuous measurement mode 1: 10Hz, measure 10 times per second,
72// (4) Continuous measurement mode 2: 20Hz, measure 20 times per second,
73// (5) Continuous measurement mode 3: 50Hz, measure 50 times per second,
74// (6) Continuous measurement mode 4: 100Hz, measure 100 times per second,
75// (7) Self-test mode
76enum AK09918_mode_type_t {
77 AK09918_POWER_DOWN = 0x00,
78 AK09918_NORMAL = 0x01,
79 AK09918_CONTINUOUS_10HZ = 0x02,
80 AK09918_CONTINUOUS_20HZ = 0x04,
81 AK09918_CONTINUOUS_50HZ = 0x06,
82 AK09918_CONTINUOUS_100HZ = 0x08,
83 AK09918_SELF_TEST = 0x10, // ignored by switchMode() and initialize(), call selfTest() to use this mode
84};
85
86enum AK09918_err_type_t {
87 AK09918_ERR_OK = 0, // ok
88 AK09918_ERR_DOR = 1, // data skipped
89 AK09918_ERR_NOT_RDY = 2, // not ready
90 AK09918_ERR_TIMEOUT = 3, // read/write timeout
91 AK09918_ERR_SELFTEST_FAILED = 4, // self test failed
92 AK09918_ERR_OVERFLOW = 5, // sensor overflow, means |x|+|y|+|z| >= 4912uT
93 AK09918_ERR_WRITE_FAILED = 6, // fail to write
94 AK09918_ERR_READ_FAILED = 7, // fail to read
95
96};
