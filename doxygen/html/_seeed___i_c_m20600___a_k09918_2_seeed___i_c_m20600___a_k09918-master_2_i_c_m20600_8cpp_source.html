38void ICM20600::setPowerMode(icm20600_power_type_t mode) {
39 uint8_t data_pwr1;
40 uint8_t data_pwr2 = 0x00;
41 uint8_t data_gyro_lp;
42 I2Cdev::readByte(_addr, ICM20600_PWR_MGMT_1, _buffer);
43 data_pwr1 = _buffer[0];
44 data_pwr1 &= 0x8f; // 0b10001111
45 I2Cdev::readByte(_addr, ICM20600_GYRO_LP_MODE_CFG, _buffer);
46 data_gyro_lp = _buffer[0];
47 // When set to ‘1’ low-power gyroscope mode is enabled. Default setting is 0
48 data_gyro_lp &= 0x7f; // 0b01111111
49 switch (mode) {
50 case ICM_SLEEP_MODE:
51 data_pwr1 |= 0x40; // set 0b01000000
52 break;
53
54 case ICM_STANDYBY_MODE:
55 data_pwr1 |= 0x10; // set 0b00010000
56 data_pwr2 = 0x38; // 0x00111000 disable acc
57 break;
58
59 case ICM_ACC_LOW_POWER:
60 data_pwr1 |= 0x20; // set bit5 0b00100000
61 data_pwr2 = 0x07; //0x00000111 disable gyro
62 break;
63
64 case ICM_ACC_LOW_NOISE:
65 data_pwr1 |= 0x00;
66 data_pwr2 = 0x07; //0x00000111 disable gyro
67 break;
68
69 case ICM_GYRO_LOW_POWER:
70 data_pwr1 |= 0x00; // dont set bit5 0b00000000
71 data_pwr2 = 0x38; // 0x00111000 disable acc
72 data_gyro_lp |= 0x80;
73 break;
74
75 case ICM_GYRO_LOW_NOISE:
76 data_pwr1 |= 0x00;
77 data_pwr2 = 0x38; // 0x00111000 disable acc
78 break;
79
80 case ICM_6AXIS_LOW_POWER:
81 data_pwr1 |= 0x00; // dont set bit5 0b00100000
82 data_gyro_lp |= 0x80;
83 break;
84
85 case ICM_6AXIS_LOW_NOISE:
86 data_pwr1 |= 0x00;
87 break;
88
89 default:
90 break;
91 }
92 I2Cdev::writeByte(_addr, ICM20600_PWR_MGMT_1, data_pwr1);
93 I2Cdev::writeByte(_addr, ICM20600_PWR_MGMT_2, data_pwr2);
94 I2Cdev::writeByte(_addr, ICM20600_GYRO_LP_MODE_CFG, data_gyro_lp);
95}
