152// ================================================================
153// === INITIAL SETUP ===
154// ================================================================
155
156void setup() {
157 // join I2C bus (I2Cdev library doesn't do this automatically)
158 Wire.begin();
159
160 // initialize serial communication
161 // (115200 chosen because it is required for Teapot Demo output, but it's
162 // really up to you depending on your project)
163 Serial.begin(115200);
164 //while (!Serial); // wait for Leonardo enumeration, others continue immediately
165
166 // NOTE: 8MHz or slower host processors, like the Teensy @ 3.3v or Ardunio
167 // Pro Mini running at 3.3v, cannot handle this baud rate reliably due to
168 // the baud timing being too misaligned with processor ticks. You must use
169 // 38400 or slower in these cases, or use some kind of external separate
170 // crystal solution for the UART timer.
171
172 // initialize device
173 Serial.println(F("Initializing I2C devices..."));
174 mpu.initialize();
175
176 // verify connection
177 Serial.println(F("Testing device connections..."));
178 Serial.println(mpu.testConnection() ? F("MPU6050 connection successful") : F("MPU6050 connection failed"));
179
180 // wait for ready
181 Serial.println(F("\nSend any character to begin DMP programming and demo: "));
182 while (Serial.available() && Serial.read()); // empty buffer
183 while (!Serial.available()); // wait for data
184 while (Serial.available() && Serial.read()); // empty buffer again
185
186 // load and configure the DMP
187 Serial.println(F("Initializing DMP..."));
188 devStatus = mpu.dmpInitialize();
189
190 // make sure it worked (returns 0 if so)
191 if (devStatus == 0) {
192 // turn on the DMP, now that it's ready
193 Serial.println(F("Enabling DMP..."));
194 mpu.setDMPEnabled(true);
195
196 // enable Arduino interrupt detection
197 Serial.println(F("Enabling interrupt detection (Arduino external interrupt 0)..."));
198 attachInterrupt(0, dmpDataReady, RISING);
199 mpuIntStatus = mpu.getIntStatus();
200
201 // set our DMP Ready flag so the main loop() function knows it's okay to use it
202 Serial.println(F("DMP ready! Waiting for first interrupt..."));
203 dmpReady = true;
204
205 // get expected DMP packet size for later comparison
206 packetSize = mpu.dmpGetFIFOPacketSize();
207 } else {
208 // ERROR!
209 // 1 = initial memory load failed
210 // 2 = DMP configuration updates failed
211 // (if it's going to break, usually the code will be 1)
212 Serial.print(F("DMP Initialization failed (code "));
213 Serial.print(devStatus);
214 Serial.println(F(")"));
215 }
216
217 // configure LED for output
218 pinMode(LED_PIN, OUTPUT);
219}

223// ================================================================
224// === MAIN PROGRAM LOOP ===
225// ================================================================
226
227void loop() {
228 // if programming failed, don't try to do anything
229 if (!dmpReady) {
230 return;
231 }
232
233 // wait for MPU interrupt or extra packet(s) available
234 while (!mpuInterrupt && fifoCount < packetSize) {
235 // other program behavior stuff here
236 // .
237 // .
238 // .
239 // if you are really paranoid you can frequently test in between other
240 // stuff to see if mpuInterrupt is true, and if so, "break;" from the
241 // while() loop to immediately process the MPU data
242 // .
243 // .
244 // .
245 }
246
247 // reset interrupt flag and get INT_STATUS byte
248 mpuInterrupt = false;
249 mpuIntStatus = mpu.getIntStatus();
250
251 // get current FIFO count
252 fifoCount = mpu.getFIFOCount();
253
254 // check for overflow (this should never happen unless our code is too inefficient)
255 if ((mpuIntStatus & 0x10) || fifoCount == 1024) {
256 // reset so we can continue cleanly
257 mpu.resetFIFO();
258 Serial.println(F("FIFO overflow!"));
259
260 // otherwise, check for DMP data ready interrupt (this should happen frequently)
261 } else if (mpuIntStatus & 0x02) {
262 // wait for correct available data length, should be a VERY short wait
263 while (fifoCount < packetSize) {
264 fifoCount = mpu.getFIFOCount();
265 }
266
267 // read a packet from FIFO
268 mpu.getFIFOBytes(fifoBuffer, packetSize);
269
270 // track FIFO count here in case there is > 1 packet available
271 // (this lets us immediately read more without waiting for an interrupt)
272 fifoCount -= packetSize;
273
274 #ifdef OUTPUT_READABLE_QUATERNION
275 // display quaternion values in easy matrix form: w x y z
276 mpu.dmpGetQuaternion(&q, fifoBuffer);
277 Serial.print("quat\t");
278 Serial.print(q.w);
279 Serial.print("\t");
280 Serial.print(q.x);
281 Serial.print("\t");
282 Serial.print(q.y);
283 Serial.print("\t");
284 Serial.println(q.z);
285 #endif
286
287 #ifdef OUTPUT_READABLE_EULER
288 // display Euler angles in degrees
289 mpu.dmpGetQuaternion(&q, fifoBuffer);
290 mpu.dmpGetEuler(euler, &q);
291 Serial.print("euler\t");
292 Serial.print(euler[0] * 180 / M_PI);
293 Serial.print("\t");
294 Serial.print(euler[1] * 180 / M_PI);
295 Serial.print("\t");
296 Serial.println(euler[2] * 180 / M_PI);
297 #endif
298
299 #ifdef OUTPUT_READABLE_YAWPITCHROLL
300 // display Euler angles in degrees
301 mpu.dmpGetQuaternion(&q, fifoBuffer);
302 mpu.dmpGetGravity(&gravity, &q);
303 mpu.dmpGetYawPitchRoll(ypr, &q, &gravity);
304 Serial.print("ypr\t");
305 Serial.print(ypr[0] * 180 / M_PI);
306 Serial.print("\t");
307 Serial.print(ypr[1] * 180 / M_PI);
308 Serial.print("\t");
309 Serial.println(ypr[2] * 180 / M_PI);
310 #endif
311
312 #ifdef OUTPUT_READABLE_REALACCEL
313 // display real acceleration, adjusted to remove gravity
314 mpu.dmpGetQuaternion(&q, fifoBuffer);
315 mpu.dmpGetAccel(&aa, fifoBuffer);
316 mpu.dmpGetGravity(&gravity, &q);
317 mpu.dmpGetLinearAccel(&aaReal, &aa, &gravity);
318 Serial.print("areal\t");
319 Serial.print(aaReal.x);
320 Serial.print("\t");
321 Serial.print(aaReal.y);
322 Serial.print("\t");
323 Serial.println(aaReal.z);
324 #endif
325
326 #ifdef OUTPUT_READABLE_WORLDACCEL
327 // display initial world-frame acceleration, adjusted to remove gravity
328 // and rotated based on known orientation from quaternion
329 mpu.dmpGetQuaternion(&q, fifoBuffer);
330 mpu.dmpGetAccel(&aa, fifoBuffer);
331 mpu.dmpGetGravity(&gravity, &q);
332 mpu.dmpGetLinearAccelInWorld(&aaWorld, &aaReal, &q);
333 Serial.print("aworld\t");
334 Serial.print(aaWorld.x);
335 Serial.print("\t");
336 Serial.print(aaWorld.y);
337 Serial.print("\t");
338 Serial.println(aaWorld.z);
339 #endif
340
341 #ifdef OUTPUT_TEAPOT
342 // display quaternion values in InvenSense Teapot demo format:
343 teapotPacket[2] = fifoBuffer[0];
344 teapotPacket[3] = fifoBuffer[1];
345 teapotPacket[4] = fifoBuffer[4];
346 teapotPacket[5] = fifoBuffer[5];
347 teapotPacket[6] = fifoBuffer[8];
348 teapotPacket[7] = fifoBuffer[9];
349 teapotPacket[8] = fifoBuffer[12];
350 teapotPacket[9] = fifoBuffer[13];
351 Serial.write(teapotPacket, 14);
352 teapotPacket[11]++; // packetCount, loops at 0xFF on purpose
353 #endif
354
355 // blink LED to indicate activity
356 blinkState = !blinkState;
357 digitalWrite(LED_PIN, blinkState);
358 }
