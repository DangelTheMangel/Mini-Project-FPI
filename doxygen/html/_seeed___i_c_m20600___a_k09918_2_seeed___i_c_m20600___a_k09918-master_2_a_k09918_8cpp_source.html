87AK09918_err_type_t AK09918::getRawData(int32_t* axis_x, int32_t* axis_y, int32_t* axis_z) {
88 if (_mode == AK09918_NORMAL) {
89 AK09918::switchMode(AK09918_NORMAL);
90 bool is_end = false;
91 int count = 0;
92 while (!is_end) {
93 if (AK09918::_getRawMode() == 0x00) {
94 is_end = true;
95 }
96 if (count >= 15) {
97 return AK09918_ERR_TIMEOUT;
98 }
99 count ++;
100 delay(1);
101 }
102 }
103
104
105 if (!I2Cdev::readBytes(_addr, AK09918_HXL, 8, _buffer)) {
106 return AK09918_ERR_READ_FAILED;
107 } else {
108 *axis_x = (int16_t)(_buffer[1] << 8 | _buffer[0]);
109 *axis_y = (int16_t)(_buffer[3] << 8 | _buffer[2]);
110 *axis_z = (int16_t)(_buffer[5] << 8 | _buffer[4]);

122AK09918_err_type_t AK09918::switchMode(AK09918_mode_type_t mode) {
123 if (mode == AK09918_SELF_TEST) {
124 return AK09918_ERR_WRITE_FAILED;
125 }
126 _mode = mode;
127 if (!I2Cdev::writeByte(_addr, AK09918_CNTL2, mode)) {
128 return AK09918_ERR_WRITE_FAILED;
129 }
130 return AK09918_ERR_OK;
131}

133// 1.Set Power-down mode. (MODE[4:0] bits = “00000”)
134// 2.Set Self-test mode. (MODE[4:0] bits = “10000”)
135// 3.Check Data Ready or not by polling DRDY bit of ST1 register.
136// 4.When Data Ready, proceed to the next step. Read measurement data. (HXL to HZH)
137AK09918_err_type_t AK09918::selfTest() {
138 int32_t axis_x, axis_y, axis_z;
139 bool is_end = false;
140 AK09918_err_type_t err;
141 if (!I2Cdev::writeByte(_addr, AK09918_CNTL2, AK09918_POWER_DOWN)) {
142 return AK09918_ERR_WRITE_FAILED;
143 }
144 delay(1);
145 if (!I2Cdev::writeByte(_addr, AK09918_CNTL2, AK09918_SELF_TEST)) {
146 return AK09918_ERR_WRITE_FAILED;
147 }
148
149 while (!is_end) {
150 err = AK09918::isDataReady();
151 if (err == AK09918_ERR_OK) {
152 is_end = true;
153 } else if (err == AK09918_ERR_READ_FAILED) {
154 return AK09918_ERR_READ_FAILED;
155 }
156 delay(1);
157 }
158
159 // read data and check
160 if (!I2Cdev::readBytes(_addr, AK09918_HXL, 8, _buffer)) {
161 return AK09918_ERR_READ_FAILED;
162 } else {
163 axis_x = (int32_t)((((int16_t)_buffer[1]) << 8) | _buffer[0]);
164 axis_y = (int32_t)((((int16_t)_buffer[3]) << 8) | _buffer[2]);
165 axis_z = (int32_t)((((int16_t)_buffer[5]) << 8) | _buffer[4]);
166

169 // Serial.print("\tY:");
170 // Serial.print(axis_y);
171 // Serial.print("\tZ:");
172 // Serial.println(axis_z);
173
174 if ((axis_x >= -200) && (axis_x <= 200) && (axis_y >= -200) && (axis_y <= 200) && \
175 (axis_z >= -1000) && (axis_z <= -150)) {
176 return AK09918_ERR_OK;

178 return AK09918_ERR_SELFTEST_FAILED;

191String AK09918::strError(AK09918_err_type_t err) {
192 String result;
193 switch (err) {
194 case AK09918_ERR_OK:
195 result = "AK09918_ERR_OK: OK";
196 break;
197
198 case AK09918_ERR_DOR:
199 result = "AK09918_ERR_DOR: Data skipped";
200 break;
201
202 case AK09918_ERR_NOT_RDY:
203 result = "AK09918_ERR_NOT_RDY: Not ready";
204 break;
205
206 case AK09918_ERR_TIMEOUT:
207 result = "AK09918_ERR_TIMEOUT: Timeout";
208 break;
209
210 case AK09918_ERR_SELFTEST_FAILED:
211 result = "AK09918_ERR_SELFTEST_FAILED: Self test failed";
212 break;
213
214 case AK09918_ERR_OVERFLOW:
215 result = "AK09918_ERR_OVERFLOW: Sensor overflow";
216 break;
217
218 case AK09918_ERR_WRITE_FAILED:
219 result = "AK09918_ERR_WRITE_FAILED: Fail to write";
220 break;
221
222 case AK09918_ERR_READ_FAILED:
223 result = "AK09918_ERR_READ_FAILED: Fail to read";
224 break;
225
226 default:
227 result = "Unknown Error";
228 break;
229 }
