94s32 CY8C_IIC_OPRTS::IIC_read_byte(u8 reg,u8* byte)
95{
96 u8 timeout_count=0;
97 Wire.beginTransmission(_IIC_ADDR);
98 Wire.write(reg);
99 Wire.endTransmission(false);
100
101 Wire.requestFrom(_IIC_ADDR,(u8)1);
102 while(1!=Wire.available())
103 {
104 timeout_count++;
105 if(timeout_count>10) return -1;
106 delay(1);
107 }
108 *byte=Wire.read();
109 return 0;
110
111}

113void CY8C_IIC_OPRTS::IIC_read_16bit(u8 start_reg,u16 *value)
114{
115 u8 val=0;
116 *value=0;
117 Wire.beginTransmission(_IIC_ADDR);
118 Wire.write(start_reg);
119 Wire.endTransmission(false);
120
121 Wire.requestFrom(_IIC_ADDR,sizeof(u16));
122 while(sizeof(u16)!=Wire.available());
123 val=Wire.read();
124 *value|=(u16)val<<8;
125 val=Wire.read();
126 *value|=val;
127}
