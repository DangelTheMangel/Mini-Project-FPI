138void getTiltHeading(void) {
139 float pitch = asin(-Axyz[0]);
140 float roll = asin(Axyz[1] / cos(pitch));
141
142 float xh = Mxyz[0] * cos(pitch) + Mxyz[2] * sin(pitch);
143 float yh = Mxyz[0] * sin(roll) * sin(pitch) + Mxyz[1] * cos(roll) - Mxyz[2] * sin(roll) * cos(pitch);
144 float zh = -Mxyz[0] * cos(roll) * sin(pitch) + Mxyz[1] * sin(roll) + Mxyz[2] * cos(roll) * cos(pitch);
145 tiltheading = 180 * atan2(yh, xh) / PI;
146 if (yh < 0) {
147 tiltheading += 360;
148 }
149}

257void getCompass_Data(void) {
258 I2C_M.writeByte(MPU9150_RA_MAG_ADDRESS, 0x0A, 0x01); //enable the magnetometer
259 delay(10);
260 I2C_M.readBytes(MPU9150_RA_MAG_ADDRESS, MPU9150_RA_MAG_XOUT_L, 6, buffer_m);
261
262 mx = ((int16_t)(buffer_m[1]) << 8) | buffer_m[0] ;
263 my = ((int16_t)(buffer_m[3]) << 8) | buffer_m[2] ;
264 mz = ((int16_t)(buffer_m[5]) << 8) | buffer_m[4] ;
265
266 Mxyz[0] = (double) mx * 1200 / 4096;
267 Mxyz[1] = (double) my * 1200 / 4096;
268 Mxyz[2] = (double) mz * 1200 / 4096;
269}
